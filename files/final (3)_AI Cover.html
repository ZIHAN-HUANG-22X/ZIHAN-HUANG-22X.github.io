def clean_text(text):
    """
    清洗微博文本：
    - 去除链接
    - 去除@用户
    - 去除话题外层符号但保留话题文字
    - 去除多余空格
    """
    if pd.isna(text):
        return ""
    
    text = str(text)
    
    # 去除链接
    text = re.sub(r"http[s]?://\S+", " ", text)
    text = re.sub(r"t\.cn/\S+", " ", text)
    
    # 去除微博视频链接中的 Lxxx
    text = re.sub(r"L[0-9A-Za-z_\-]+的微博视频", " ", text)
    
    # 去除@用户
    text = re.sub(r"@\S+", " ", text)
    
    # 保留话题文字，去掉##
    text = re.sub(r"#([^#]+)#", r"\1", text)
    
    # 去掉多余空白
    text = re.sub(r"\s+", " ", text).strip()
    
    return text


def parse_count(x):
    """
    将转发数、评论数、点赞数转成数值。
    可处理：'1.2万'、'转发'、'评论'、'赞'、空值。
    """
    if pd.isna(x):
        return 0
    
    x = str(x).strip().replace(",", "")
    
    if x in ["", "转发", "评论", "赞", "点赞"]:
        return 0
    
    try:
        if "万" in x:
            return float(x.replace("万", "")) * 10000
        else:
            num = re.sub(r"[^\d.]", "", x)
            return float(num) if num != "" else 0
    except:
        return 0


def extract_base_url(url):
    """
    统一微博链接格式，去除参数。
    """
    if pd.isna(url):
        return ""
    
    url = str(url).strip()
    url = url.split("?")[0]
    return url


def minmax_norm(series):
    """
    0-1归一化。
    如果最大值等于最小值，则返回0。
    """
    series = pd.to_numeric(series, errors="coerce").fillna(0)
    min_val = series.min()
    max_val = series.max()
    
    if max_val == min_val:
        return series * 0
    
    return (series - min_val) / (max_val - min_val)


def keyword_score(text, word_list):
    """
    计算某文本中关键词出现次数。
    """
    text = str(text)
    score = 0
    
    for word in word_list:
        score += text.count(word)
    
    return score

from snownlp import SnowNLP

def get_sentiment(text):
    """
    SnowNLP情感分析。
    返回0-1之间的情感得分，越接近1越正向。
    """
    text = str(text)
    
    if text.strip() == "":
        return np.nan
    
    try:
        return SnowNLP(text).sentiments
    except:
        return np.nan

def is_valid_word(word):

    word = str(word).strip()

    if not word:
        return False

    if len(word) <= 1:
        return False

    if word in stopwords:
        return False

    if word.isdigit():
        return False

    if re.fullmatch(r"[A-Za-z]+", word):
        return False

    if re.fullmatch(r"[A-Za-z0-9]+", word):
        return False

    return True

def get_typical_comments(df, topic_col, n=10, sentiment_order="high"):
    """
    提取某一主题下典型评论。
    默认优先选择点赞较高、长度适中的评论。
    sentiment_order:
    - high：优先正向评论
    - low：优先负向评论
    """
    temp = df[df[topic_col] == 1].copy()
    
    temp = temp[temp["comment_clean"].str.len().between(5, 100)]
    
    if sentiment_order == "high":
        temp = temp.sort_values(
            by=["likes", "comment_sentiment"],
            ascending=[False, False]
        )
    elif sentiment_order == "low":
        temp = temp.sort_values(
            by=["likes", "comment_sentiment"],
            ascending=[False, True]
        )
    else:
        temp = temp.sort_values(
            by=["likes"],
            ascending=False
        )
    
    return temp[[
        "comment_clean",
        "comment_sentiment",
        "likes",
        "comment_level"
    ]].head(n)
